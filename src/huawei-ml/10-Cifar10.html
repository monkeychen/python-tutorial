#输入图像大小为（32, 32, 3）
def CNN_classification_model(input_size = x_train.shape[1:]):
    model = Sequential() 
#卷积层，32个卷积核，卷积核大小为3*3，padding=“same”表示通过padding使输入输出图像大小一致
    model.add(Conv2D(32, (3, 3), padding='same', 
              input_shape=input_size))     #卷积核大小：3*3*3   特征图大小：32*32*32
#激活函数
    model.add(Activation('relu')) 
    model.add(Conv2D(32, (3, 3))) #卷积核大小：3*3*32   特征图大小：30*30*32
    model.add(Activation('relu')) 
#池化层
    model.add(MaxPooling2D(pool_size=(2, 2))) #特征图大小：15*15*32
#dropout层：随机丢弃25%的神经元
    model.add(Dropout(0.25)) 

    model.add(Conv2D(64, (3, 3), padding='same')) #卷积核大小：3*3*32   特征图大小：15*15*64
    model.add(Activation('relu')) 
    model.add(Conv2D(64, (3, 3))) #卷积核大小：3*3*64   特征图大小：13*13*64
    model.add(Activation('relu')) 
    model.add(MaxPooling2D(pool_size=(2, 2))) #特征图大小：6*6*64
    model.add(Dropout(0.25))  

    #将特征图送进全连接网络之前，应先将其展开为列向量
    model.add(Flatten()) 
	#全连接层
    model.add(Dense(512)) 
    model.add(Activation('relu')) 
    model.add(Dropout(0.5)) 
#最后输出10的数字，代表每个类别的输出概率
    model.add(Dense(num_classes)) 
	#softmax将每个类别的输出分数映射为分类概率
    model.add(Activation('softmax')) 
    
	#优化器
    opt = keras.optimizers.RMSprop(lr=0.0001, decay=1e-6)
    
	#模型编译
    model.compile(loss='categorical_crossentropy', optimizer=opt, metrics=['accuracy']) #sparse_categorical_crossentropy
    return model
